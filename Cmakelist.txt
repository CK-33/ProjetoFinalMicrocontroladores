cmake_minimum_required(VERSION 3.13)

# Importa o SDK do Pico (PICO_SDK_PATH deve estar no ambiente ou em cmake cache)
include($ENV{PICO_SDK_PATH}/external/pico_sdk_import.cmake)

project(acs_rp2040 C CXX ASM)
set(CMAKE_C_STANDARD   11)
set(CMAKE_CXX_STANDARD 17)

pico_sdk_init()

add_executable(acs
    acs.c
)

# Gera o header ws2812.pio.h a partir do arquivo PIO
# ws2812.pio deve estar na mesma pasta (copie de pico-examples/pio/ws2812/)
pico_generate_pio_header(acs ${CMAKE_CURRENT_LIST_DIR}/ws2812.pio)

target_link_libraries(acs
    pico_stdlib
    pico_multicore
    hardware_adc
    hardware_pwm
    hardware_pio
    hardware_timer
    hardware_clocks
    hardware_gpio
    m                   # libm — necessária para fabsf()
)

# Saídas: .uf2 (drag-and-drop), .elf (debug), .map
pico_add_extra_outputs(acs)

# Habilita saída USB serial (útil para debug via printf)
pico_enable_stdio_usb(acs  1)
pico_enable_stdio_uart(acs 0)
